Arrow Function
funciones flecha

Muchas veces hemos anidado funciones en las que nos gustaría para preservar el contexto de thisde su ámbito léxico. Un ejemplo se muestra a continuación:


function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character; 
        // Cannot read property 'name' of undefined
    });
};

Una solución común a este problema es almacenar el contexto de la thisutilización de una variable: 


function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    var that = this; // Store the context of this
    return arr.map(function (character) {
        return that.name + character;
    });
};

También podemos pasar en el contexto adecuado de this:


function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character;
    }, this);
};

Así como se unen el contexto:


function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character;
    }.bind(this));
};

Usando Flecha funciones , el valor léxico de thisque no es la sombra y podemos volver a escribir lo anterior como se muestra:


function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(character => this.name + character);
};
